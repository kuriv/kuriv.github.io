### 基本语法

附例一：

```php
<?php

declare(strict_types = 1);

$foo = [2, 4, 5];
$bar = [0, 1, 2];

var_dump($foo + $bar); // array(3) { [0]=> int(2) [1]=> int(4) [2]=> int(5) }

```

附例二：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => '2', 'bar' => '04', 'baz' => '05'];

var_dump($foo + $bar); // array(3) { ["foo"]=> string(3) "cat" ["bar"]=> string(3) "dog" ["baz"]=> string(2) "05" }

```

附例三：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => '2', 'bar' => '04', 'baz' => '05'];

var_dump($bar + $foo); // array(3) { ["foo"]=> string(1) "2" ["bar"]=> string(2) "04" ["baz"]=> string(2) "05" }

```

附例四：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['cat', 'dog'];

var_dump($foo == $bar); // bool(false)

```

附例五：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => 0, 'bar' => 'dog'];

var_dump($foo == $bar); // bool(true)

```

附例六：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => 0, 'bar' => 'dog'];

var_dump($foo === $bar); // bool(false)

```

附例七：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['bar' => 'dog', 'foo' => 'cat'];

var_dump($foo === $bar); // bool(false)

```

附例八：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['cat', 'dog'];

var_dump($foo != $bar); // bool(true)

```

附例九：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => 0, 'bar' => 'dog'];

var_dump($foo != $bar); // bool(false)

```

附例十：

```php
<?php

declare(strict_types = 1);

$foo = ['foo' => 'cat', 'bar' => 'dog'];
$bar = ['foo' => 0, 'bar' => 'dog'];

var_dump($foo !== $bar); // bool(true)

```

```php
<?php
```
